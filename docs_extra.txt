1. Architecture Overview
HabitFlow uses a 100% client-side, local-first data architecture. There is no traditional backend server. The database lives inside the user's browser via the IndexedDB Web API, managed through Dexie.js О“ГҮГ¶ a battle-tested, typed IndexedDB wrapper. All business logic runs on the frontend in TypeScript service modules.
2. Entity Relationship Map
The six core collections and their relationships. IndexedDB does not enforce foreign keys natively О“ГҮГ¶ referential integrity is enforced at the Dexie service layer.

4. Zod Validation Schemas
Every collection has a Zod schema. All data passing into IndexedDB is parsed through these schemas. Parse failures are caught at the service layer О“ГҮГ¶ invalid data never reaches the database.

Indexes in Dexie are declared in the .stores() schema string. Only frequently-queried fields should be indexed О“ГҮГ¶ over-indexing slows writes and increases storage.
6. Service Layer
Each collection has a dedicated service module that encapsulates all database operations. Components and hooks never call db directly О“ГҮГ¶ they go through the service layer. This enforces validation, referential integrity, and business rules in one place.

7. Migration Strategy
Dexie's built-in version system handles schema evolution. Every production schema change requires a new version number with an explicit upgrade function. Skipping a version is forbidden.
8. Transaction Patterns
Multi-table writes must always be wrapped in a Dexie transaction. A transaction is atomic О“ГҮГ¶ if any step fails, all changes roll back. Never perform related writes outside a transaction.
9. Data Export & Import Schema
All user data can be exported as a single JSON file and re-imported on any device. The export schema is versioned independently of the database schema to allow cross-version restores.
10. Analytics Query Patterns
Analytics data is never stored pre-computed. Every chart is generated on demand by querying IndexedDB and transforming the results in application memory. For heavy computations, a Web Worker is used to avoid blocking the UI thread.

export async function checkIn(
  habitId: string,
  date: string,      // YYYY-MM-DD
  value: number,
  note?: string
): Promise<HabitLog> {
  // Check for existing log (compound index lookup О“ГҮГ¶ O(log n))
  const existing = await db.habitLogs
    .where('[habitId+date]').equals([habitId, date])
    .first()

  const habit = await db.habits.get(habitId)
  if (!habit) throw new Error(`Habit ${habitId} not found`)

  const completed = habit.targetValue
    ? value >= habit.targetValue ? 1 : 0
    : value > 0 ? 1 : 0

  if (existing) {
    // Update existing entry
    await db.habitLogs.update(existing.id, {
      value, completed, note, isBackdated: date !== todayISO() ? 1 : 0
    })
    return { ...existing, value, completed }
  }

  // Create new entry
  const log = HabitLogSchema.parse({
    id: generateId(), habitId, date, value, completed,
    note, isBackdated: date !== todayISO() ? 1 : 0,
    timeOfDay: currentTime(),  // HH:MM
    createdAt: new Date(),
  })
  await db.habitLogs.add(log)

  // Award XP in the same transaction
  await awardXP(habitId, value, completed)
  return log
}

export async function getStreak(habitId: string): Promise<{ current: number; best: number }> {

    .where('habitId').equals(habitId)

// Generate 52-week heatmap data for a habit
export async function getHeatmapData(habitId: string): Promise<HeatmapCell[]> {
  const start = subDays(new Date(), 364)   // 52 weeks back
  const end   = new Date()

  const logs = await db.habitLogs
    .where('habitId').equals(habitId)
    .filter(l => isAfter(parseISO(l.date), start))
    .toArray()

  const logMap = new Map(logs.map(l => [l.date, l]))
  const habit  = await db.habits.get(habitId)

  // Build cell for every day in range
  return eachDayOfInterval({ start, end }).map(day => {
    const dateStr = format(day, 'yyyy-MM-dd')
    const log     = logMap.get(dateStr)
    const pct     = log && habit?.targetValue
      ? Math.min(100, Math.round((log.value / habit.targetValue) * 100))
      : log?.completed ? 100 : 0
    return { date: dateStr, value: pct, raw: log?.value ?? 0 }
  })
}

// shared/utils/streakCalc.ts О“ГҮГ¶ pure function, testable

export function calcStreakFromLogs(
  logs: Pick<HabitLog, 'date' | 'completed'>[],
  frequency: Frequency,
  graceDays: number
): { current: number; best: number } {
  const completedDates = new Set(logs.filter(l => l.completed).map(l => l.date))
  const scheduledDays  = getScheduledDays(frequency)  // days habit should be done

  let current = 0, best = 0, streak = 0, missesThisWeek = 0

  // Walk backwards from yesterday (today is still in progress)
  let cursor = subDays(new Date(), 1)
  while (isAfter(cursor, subDays(new Date(), 365))) {
    const dateStr   = format(cursor, 'yyyy-MM-dd')
    const isScheduled = scheduledDays.has(cursor.getDay())

    if (isScheduled) {
      if (completedDates.has(dateStr)) {
        streak++
        missesThisWeek = 0
      } else {
        if (missesThisWeek < graceDays) {
          missesThisWeek++   // grace day consumed О“ГҮГ¶ streak continues
        } else {
          best = Math.max(best, streak)
          if (current === 0) current = streak
          streak = 0         // streak broken
        }
      }
    }
    cursor = subDays(cursor, 1)
  }

  best = Math.max(best, streak)
  return { current: current || streak, best }
}
